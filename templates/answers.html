n = input("input a decimal number: ")
def makebinary(n):
  n = int(n)
  output = ""
  if(n == 0):
    output = "0"
  while n >= 1:
    output = str(n%2) + output 
    n //=2
  print(output)

def makehex(n):
  n = int(n)
  output = ""
  if(n == 0):
    output = "0"
  while n >= 1:
    if(n%16 == 10):
      output = "A" + output 
    elif(n%16 == 11):
      output = "B" + output
    elif(n%16 == 12):
      output = "C" + output 
    elif(n%16 == 13):
      output = "D" + output 
    elif(n%16 == 14):
      output = "E" + output 
    elif(n%16 == 15):
      output = "F" + output 
    else:
      output = str(n%16) + output 
    n //=16
  print(output)
